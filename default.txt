using System.Xml.Linq;

namespace MemoryGame
{
    class Program
    {
        public static void Main()
        {
            List<int> sequence = Console
                .ReadLine()
                .Split()
                .Select(int.Parse)
                .ToList();
            string untilEnd;
            int moves = 0;
            while ((untilEnd = Console.ReadLine()) != "end")
            {
                moves++;
                List<int> indexes = Console
                    .ReadLine()
                    .Split()
                    .Select(int.Parse)
                    .ToList();
                int index1 = indexes[0];
                int index2 = indexes[1];
                if ((index1 == index2) ||
                    (index1 < 0 || index1 > sequence.Count) ||
                    (index2 < 0 || index2 > sequence.Count))
                {
                    Console.WriteLine($"-{moves}a");
                    Console.WriteLine("Invalid input! Adding additional elements to the board");
                }
                else if (sequence[index1] == sequence[index2])
                {
                    Console.WriteLine($"Congrats! You have found matching elements - {sequence[index1]}!");
                    sequence.RemoveAt(index1);
                    sequence.RemoveAt(index2);
                }
            }
        }
    }
}
